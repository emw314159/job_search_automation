            if (jQuery(this).val() == "" || jQuery(this).val() == jQuery(this)[0].title) {

            }
        });
        jQuery("#keywords").blur();

    function encode(val) {

            .replace(/\*/gi, "{asterisk}")

            .replace(/:/gi, "{colon}")

            .replace(/\]/gi, "{rightbracket}")
